Bouncing ball problem

rem: Bouncing ball problem

Source: OpenOCL - Open Optimal Control Library. 

set:
  phase = bc,ac		# before contact, after contact

par:
  tf = 1

var:
  tlen[phase]

dyn:
  t[phase]		# states
  s[phase]
  v[phase]
  F[phase]		# control
  J[phase]		# criterion

lim:
  0.1 <= tlen[phase] <= 10
  0 <= F['bc'] <= 0

t=t0:
  t['bc'] = 0
  s['bc'] = 1
  v['bc'] = 0
  J['bc'] = 0

t=tf:
  s['bc'] = 0
  s['ac'] = 1

exp:
  TimeDot == tlen[phase] / tf

equ:
  t´[phase] / TimeDot == 1
  s´[phase] / TimeDot == v[phase]
  v´[phase] / TimeDot == -10 + 10*F[phase]
  J´[phase] / TimeDot == sqr(F[phase])
  @linkphases -x v	# link final and initial conditions across phases for state variables, except for variable v, which is given here after
  $once.. initial(v['ac']) == -final(v['bc'])/2

obj:
  minimize final(J['ac']) using nlp
