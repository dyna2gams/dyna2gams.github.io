Moonlander problem

rem: Moonlander problem

Source: PSOPT - Optimal Control Solver, User Manual, Release 4 build 2015. 

par:
  exhaust_velocity = 2.349
  gravity = 1
  
var:
  tf

dyn:
  altitude speed mass	# state variables
  thrust		# control variable
  J			# cost criterion

lim:
  -20 <= altitude <= 20
  -20 <= speed <= 20
  0.01 <= mass <= 1
  0 <= thrust <= 1.227
  0 <= tf <= 1000

t=t0:
  altitude = 1
  speed = -0.783
  mass = 1
  J = 0
  
t=tf:
  altitude = 0
  speed = 0
  
ini:
  altitude = linspace(initial(altitude),final(altitude))
  speed = linspace(initial(speed),final(speed))
  mass = linspace(initial(mass),0.01)
  
equ:
  altitude´ == speed
  speed´ == -gravity + thrust/mass
  mass´ == -thrust/exhaust_velocity
  J´ == thrust

obj: 
  minimize final(J) using nlp
