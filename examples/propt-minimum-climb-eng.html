exp:
  rho == §AD(h*0.3048) * 0.001941
  sos == §SS(h*0.3048) / 0.3048
  Mach == v / sos
  CD0 == §CD0(Mach)
  CLalpha == §CLalpha(Mach)
  eta == §eta(Mach)
  T == §Thrust(Mach,h)
  CD == CD0 + eta * CLalpha * sqr(alpha)
  CL == CLalpha * alpha
  dynpres == 0.5 * rho * sqr(v)
  D == dynpres * S * CD
  L == dynpres * S * CL

ini:
  tf = 300

  fpa = 10/180*pi

  alpha = linspace(20/180*pi,-20/180*pi)

equ:
  h´   == v * sin(fpa)
  v´   == (T * cos(alpha) - D) / m - mmu * sin(fpa) / sqr(Re+h)
  fpa´ == (T * sin(alpha) + L) / (m * v) + cos(fpa) * (v / (Re+h) - mmu / (v * sqr(Re+h)))
  m´   == -T / (g0 * ISP)
