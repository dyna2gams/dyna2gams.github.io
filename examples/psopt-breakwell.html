Breakwell problem

rem: Breakwell problem

Source: PSOPT Optimal Control Solver, User Manual, Release 4 build 2015 

par:
  tf = 1
  
dyn:
  x v	# state variables
  u	# control variables
  J	# cost criterion
  
var: 

lim:
  -2 <= x <= 0.1
  -2 <= v <= 2
  -10 <= u <= 10

t=t0:
  J = 0
  x = 0
  v = 1

t=tf:
  x = 0 
  v = -1

ini:
  x = initial(x)
  v = initial(v)
  u = 0

equ:
  x´ == v
  v´ == u
  J´ == 0.5*u*u

obj: 
  minimize final(J) using nlp

tpa:
  ua1 
  xa{1:2}
  pa{1:2}

gms:
  scalar l / 0.1 /;
  scalar t1; t1 = 3*l 
  scalar t2; t2 = 1-3*l;
  ua1 $ (Time < t1) = -2/(3*l)*(1-Time/(3*l));
  xa1 $ (Time < t1) = l*(1-power((1-Time/(3*l)),3));
  xa2 $ (Time < t1) = power(1-Time/(3*l),2);
  pa1 $ (Time < t1) = 2/(9*l*l);
  pa2 $ (Time < t1) = 2/(3*l)*(1-Time/(3*l));
  ua1 $ (Time >= t1 and Time < t2) = 0;
  xa1 $ (Time >= t1 and Time < t2) = l;
  xa2 $ (Time >= t1 and Time < t2) = 0;
  pa1 $ (Time >= t1 and Time < t2) = 0;
  pa2 $ (Time >= t1 and Time < t2) = 0;
  ua1 $ (Time >= t2) = -2/(3*l)*(1-(1-Time)/(3*l));
  xa1 $ (Time >= t2) = l*(1-power((1-(1-Time)/(3*l)),3));
  xa2 $ (Time >= t2) = -power(1-(1-Time)/(3*l), 2) ;
  pa1 $ (Time >= t2) = -2/(9*l*l);
  pa2 $ (Time >= t2) = 2/(3*l)*(1-(1-Time)/(3*l));
