par:
  Anp = 4e+4
  anp{1:8} = {6.255 24.1 0.4027 1.511 0.3596 0.9603 0.6938 20.71}
  wnp{1:8} = {0.6837 0.7458 1.354 0.5228 1.054 0.3953 0.3246 0.7512}
  phinp{1:8} = {0.4082 1.727 -0.4019 -1.737 -2.663 -1.51 -2.364 4.73}
  gamma = 1e+5
  m = 2e+5		# [kg]
  k = 1.2e+5		# [kg/s²]
  c = 1e+5		# [kg m²/s³]
  tf = 50

dyn:
  x{1:2}
  u
  J

lim:
  -gamma <= u <= gamma

t=t0:
  x{1:2} = 0
  J = 0

exp:
  fs == -k*x1
  fc == -c*x2
  fe == Anp*sum{1:8, anp{:}*sin(wnp{:}*Time+phinp{:})}

equ:
  x1´ == x2
  x2´ == (fe + fs + fc - u)/m
  J´ == u*x2

obj:
  maximize 1e-6*final(J) using nlp
