Robot

rem: Robot

A simple robot model.

See: DyCon project 

par:
  alpha = 0.2
  delta = 1
  gamma = 1
  tf = 12

dyn:
  x{1:2}
  theta
  nu{1:2}
  omega
  u{1:2}
  J

t=t0:
  x{1:2} = -10
  theta = 0.5*pi
  nu{1:2} = 0
  omega = 0
  J = 0

equ:
  x{1:2}´ == nu{1:2}
  theta´ == omega
  nu1´ == (u1 + u2)*cos(theta)
  nu2´ == (u1 + u2)*sin(theta)
  omega´ == alpha*(u1-u2)
  J´ == sqr(u1)+sqr(u2)
  #J´ == abs(u1)+abs(u2)

obj:
  minimize final(0.5*delta*(sqr(x1)+sqr(x2)+sqr(theta)+sqr(nu1)+sqr(nu2)+sqr(omega)) + gamma*J) using dnlp

gms:
  @solveloop Method=bndobj
