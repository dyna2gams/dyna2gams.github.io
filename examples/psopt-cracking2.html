dyn:
  y{1:2}  # state variables

var:
  theta{1:3}

lim:
  0 <= y{1:2} <= 2
  0 <= theta{1:3} <= 20

ini:
  y1 = linspace(1,0.069)
  y2 = linspace(0.3,0.01)
  theta{1:3} = 0

equ:
  y1´ == -(theta1 + theta3)*y1*y1
  y2´ == theta1*y1*y1 - theta2*y2

obj:
  minimize \
  sum(obs, \
    sqr(eval(y1,data(obs,'tau'))-data(obs,'y1m')) + \
    sqr(eval(y2,data(obs,'tau'))-data(obs,'y2m')) \
    ) \
  using nlp
