Discrete Time Rocket

rem: Discrete Time Rocket

Source: ACADO Toolkit - A Toolkit for Automatic Control and Dynamic Optimization. 

run:
  time-model = discrete
  N = 200

par:
  h = 0.1
  ts = 1

dyn:
  v s m
  u
  ke	# Kinetic energy
  J

lim:
  -0.01 <= v <= 1.3
  0.001 <= m <= 1
  -0.5 <= u <= 0.5

k=k0:
  s = 0
  v = 0
  m = 1
  J = 0
  
k=kf:
  s = 1
  v = 0

ini:
  s = linspace(0,10)  
  v = 4*linspace(0,1)*(linspace(0,1)-1)
  m = 1
  u = 0.1*v*v

equ:
  s´ == s + h*v
  v´ == v + h*(u-0.02*v*v)/m
  m´ == m - h*0.01*u*u
  ke == 0.5*m*v*v 
  J´ == J + u*u

obj:
  minimize final(J) using nlp
