Vasicek Short Rate Model

rem: Vasicek Short Rate Model


run:
  simulation-method = SDE/SI
  simulation-loops-number = 32

set:
  scen = sce1:sce9

par:
  a = 0.7	# reversion speed
  b = 0.029	# long term rate
  sigma = 0.01	# volatility
  tf = 10

dyn:
  y		# yield curve
  x[scen]	# short rate
  W[scen]

t=t0:
  y = 0.025
  x[scen] = 0.025

sim:
  y´ := a*(b-y)
  W[scen] :: Wiener
  x´[scen] := a*(b-x[scen]) + sigma*W[scen]
