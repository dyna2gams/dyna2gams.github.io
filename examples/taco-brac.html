The classical brachistochrone problem

rem: The classical brachistochrone problem

See: mintoc.de 

par:
  gravity = 32.174

dyn:
  x y v a a_dot

var: 
  tf

lim:
  0 <= tf <= 1
  0 <= x <= 1
  0 <= y <= 1
  0 <= v <= 8
  0 <= a <= 1.57079327
  -10 <= a_dot <= 10

t=t0: 
  x = 0
  y = 0
  v = 0

t=tf: 
  x = 1

ini: 
  tf = 1
  x = linspace(0,1)
  y = 0
  v = 0
  a = 0.5

equ:
  x´ == v*cos(a)
  y´ == v*sin(a)
  v´ == gravity*sin(a)
  a´ == a_dot

obj: minimize tf using nlp
