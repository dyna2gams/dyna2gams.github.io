Missile terminal burn manoeuvre

rem: Missile terminal burn manoeuvre

Source: PSOPT - Optimal Control Solver, User Manual, Release 4 build 2015 

par:
  m = 1005.0	# kg
  g = 9.81	# m/s^2
  Sref = 0.3376	# m^2
  A1 = -1.9431
  A2 = -0.1499
  A3 = 0.2359
  B1 = 21.9
  B2 = 0.0
  C1 = 3.312e-9
  C2 = -1.142e-4
  C3 = 1.224

var:
  tf
  
dyn:
  # state variables
  gamma		# flight path angle
  V		# speed
  x		# horizontal position
  h		# altitude
  # control variables
  T		# Thrust
  alpha		# Angle of attack
  # constrained paths
  z{1:2}
  
lim:
  1000 <= T <= 6000
  -0.3 <= alpha <= 0.3
  -pi/2 <= gamma <= +pi/2
  200 <= V <= 310
  0 <= x <= 10000
  0 <= h <= 1900
  10 <= tf <= 100  
  -4 <= z1 <= 4
  0 <= z2 <= 1900

t=t0:
  gamma = 0.0
  V = 272.0
  x = 0.0
  h = 30.0

t=tf:
  gamma = -pi/2.0
  V = 310
  x = 10000
  h = 0
  
ini:
  gamma = linspace(0,-pi/2)
  V = linspace(272,310)
  x = linspace(0,10000)
  h = linspace(30,0)
  T = 6000
  alpha = 0

exp:
  rho == C1*h*h + C2*h + C3	# Air density
  Cd == A1*alpha*alpha + A2*alpha + A3
  Cl == B1*alpha + B2
  D == 0.5*Cd*rho*V*V*Sref	# Axial aerodynamic force
  L == 0.5*Cl*rho*V*V*Sref	# Normal aerodynamic force
  HH == 0.5*(1+tanh((7500-x)/10))

equ:
  gamma´ == (T-D)/(m*V)*sin(alpha) + L/(m*V)*cos(alpha) - g*cos(gamma)/V
  V´ == (T-D)/m*cos(alpha) - L/m*sin(alpha) - g*sin(gamma)
  x´ == V*cos(gamma)
  h´ == V*sin(gamma)
  z1 == L/(m*g)
  z2 == HH*(h-30) + (1-HH)*h  

obj:
  minimize tf using nlp
