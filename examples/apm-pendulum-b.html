run:
  default-time-steps-number = 12
  default-iterations-number = 6

par:
  m = 1
  g = 9.81
  s = 1
  tf = 5

dyn: x y v w lam

t=t0:
  x = 0
  y = -s
  v = 1
  w = 0

ini:
  lam = m*(1+s*g)/2*s^2
  x = 0
  y = -s
  v = 1
  w = 0

equ:
  m*(v*v+w*w-g*y) - 2*lam*(x*x+y*y) == 0
  x´ == v
  y´ == w
  m*v´ == -2*x*lam
  m*w´ == -m*g - 2*y*lam

obj: minimize final(y) using nlp
