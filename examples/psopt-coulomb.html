Coulomb friction problem

rem: Coulomb friction problem

Source: PSOPT - Optimal Control Solver, User Manual, Release 4 build 2015 

par:
  k1 = 0.95 
  k2 = 0.85
  mu = 1.0
  m1 = 1.1
  m2 = 1.2
  epsilon = 0.01

dyn:
  q1 q1dot q2 q2dot   # state variables
  u1 u2               # control variables
  
var:
  tf
  
lim:
  -2 <= q1 <= 2
  -20 <= q1dot <= 20
  -2 <= q2 <= 2 
  -20 <= q2dot <= 20
  -4 <= u1 <= 4
  -4 <= u2 <= 4
  1.8 <= tf <= 4
  
t=t0:
  q1 = 0
  q1dot = -1
  q2 = 0
  q2dot = -2
  
t=tf:
  q1 = 1
  q1dot = 0
  q2 = 2
  q2dot = 0
  
ini:
  q1 = linspace(0,1)
  q1dot = linspace(-1,0)
  q2 = linspace(0,2)
  q2dot = linspace(-2,0)
  u1 = 0
  u2 = 0
  tf = 4

equ:
  q1´ == q1dot;
  q1dot´ == ((-k1-k2)*q1+k2*q2-mu*tanh(q1dot/epsilon)+u1)/m1
  q2´ == q2dot
  q2dot´ == (k2*q1-k2*q2-mu*tanh(q2dot/epsilon)+u2)/m2

obj: 
  minimize tf using nlp
