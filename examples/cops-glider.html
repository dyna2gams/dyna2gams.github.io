Hang glider (COPS Problem 11)

rem: Hang glider (COPS Problem 11)

See: COPS: Large-Scale Optimization Problems 

par:
  rho = 1.13
  r_0 = 100
  u_c = 2.5
  c0 = 0.034
  c1 = 0.069662
  S = 14
  g = 9.81
  m = 100

dyn:
  x y vx vy	# state variables
  cL		# control variable

var:
  tf

lim:
  0 <= cL <= 1.4
  0 <= tf <= 200
  0 <= x <= +inf
  0 <= vx <= +inf

t=t0:
  x = 0
  y = 1000
  vx = 13.23
  vy = -1.288

t=tf:
  y = 900
  vx = 13.23
  vy = -1.288

ini:
  tf = 100
  x = linspace(0,1323)
  y = linspace(1000,900)
  vx = 13.23
  vy = -1.288
  cL = 0.7

exp:
  r == sqr(x/r_0 - 2.5)
  u == u_c*(1-r)*exp(-r)
  w == vy - u
  v == sqrt(vx*vx+w*w)
  D == 0.5*(c0+c1*cL*cL)*rho*S*v*v
  L == 0.5*cL*rho*S*v*v
  ax == (-L*w-D*vx)/v/m
  ay == (L*vx-D*w)/v/m - g

equ:
  x´ == vx
  y´ == vy
  vx´ == ax
  vy´ == ay
  cL :: spline3

obj:
  maximize final(x) using nlp

set:
  rep = [r,u,w,v,D,L,vx_dot,vy_dot]

tpa:
  report(rep)

gms:
  $$ondotL
  report('r') = r;
  report('u') = u;
  report('w') = w;
  report('v') = v;
  report('D') = D;
  report('L') = L;
  report('vx_dot') = ax;
  report('vy_dot') = ay;
  display %MODEL%.objval;
  @csvsave report(rep)
