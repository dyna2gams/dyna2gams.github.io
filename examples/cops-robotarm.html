Robot arm (COPS Problem 8)

rem: Robot arm (COPS Problem 8)

Minimize the time taken for a robot arm to travel between two points.

See: COPS: Large-Scale Optimization Problems 

par:
  L = 5

dyn:
  rho the phi
  rho_dot the_dot phi_dot
  u_rho u_the u_phi

var:
  tf

lim:
  1 <= tf <= 12
  0 <= rho <= L; -pi <= the <= pi; 0 <= phi <= pi
  -1 <= u_rho <= 1; -1 <= u_the <= 1; -1 <= u_phi <= 1

t=t0:
  rho = 4.5; the = 0; phi = pi/4
  rho_dot = 0; the_dot = 0; phi_dot = 0

t=tf:
  rho = 4.5; the = 2*pi/3; phi = pi/4
  rho_dot = 0; the_dot = 0; phi_dot = 0

ini:
  tf = 6.5
  rho = L/2; the = 0; phi = pi/2
  rho_dot = 0; the_dot = 0; phi_dot = 0
  u_rho = 0; u_the = 0; u_phi = 0

exp:
  i_the == ((L-rho)**3+rho**3)*(sin(phi))**2/3
  i_phi == ((L-rho)**3+rho**3)/3

equ:
  rho´ == rho_dot
  the´ == the_dot
  phi´ == phi_dot
  rho_dot´ == u_rho/L
  the_dot´ == u_the/i_the
  phi_dot´ == u_phi/i_phi

obj: minimize tf using nlp
