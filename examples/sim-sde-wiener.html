Simple stochastic differential equation

rem: Simple stochastic differential equation


run:
  simulation-method = SDE/SI
  simulation-loops-number = 10

set:
  scen = sce1:sce9

par:
  alpha = 0.9
  tf = 200

dyn:
  x[scen]
  W[scen]

t=t0:
  x[scen] = 0

sim:
  W[scen] :: Wiener
  x´[scen] := -alpha*x[scen] + W[scen]
