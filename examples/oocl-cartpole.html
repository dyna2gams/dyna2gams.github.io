Cartpole problem

rem: Cartpole problem

Source: OpenOCL - Open Optimal Control Library. 

par:
  g = 9.8
  cm = 1
  pm = 0.1
  phl = 0.5	# pole half length
  m = cm+pm
  pml = pm*phl	# pole mass length
  p0 = 0
  v0 = 0
  theta0 = pi
  omega0 = 0

var:
  tf

dyn:
  p		# states
  theta
  v
  omega
  F		# control
  
lim:
  0.1 <= tf <= 10
  -5 <= p <= 5
  -2*pi <= theta <= 2*pi
  -12 <= F <= 12

t=t0:
  p = p0
  v = v0
  theta = theta0
  omega = omega0

t=tf:
  p = 0
  v = 0
  theta = 0
  omega = 0

exp:
  ctheta == cos(theta)
  stheta == sin(theta)
  domega == (g*stheta + ctheta*(-F-pml*sqr(omega)*stheta)/m) / (phl*(4/3 - pm*sqr(ctheta)/m))

equ:
  p´ == v
  theta´ == omega
  v´ == (F + pml*(sqr(omega)*stheta-domega*ctheta)) / m
  omega´ == domega

ini:
  p = linspace(p0,0)
  v = linspace(v0,0)
  theta = linspace(theta0,0)
  omega = linspace(omega0,0)

obj:
  minimize tf using nlp
