Singular problem

rem: Singular problem

Source: PSOPT - Optimal Control Solver, User Manual, Release 4 build 2015. 

par:
  tf = 1

dyn:
  x{1:3}   # state variables
  u        # control variable
  J        # cost criterion
  
lim:
  -inf <= x{1:3} <= +inf
  0 <= u <= 10
  
t=t0:
  x1 = 0
  x2 = -1
  x3 = -sqrt(5)
  J = 0
  
ini:
  x1 = 0
  x2 = -1
  x3 = -sqrt(5)

equ:
  x1´ == x2
  x2´ == -x3*u + 16*Time - 8
  x3´ == u
  J´ == x1*x1 + x2*x2 + 0.0005*sqr(x2+16*Time-8-0.1*x3*u*u)
  
obj: 
  minimize final(J) using nlp
  
gms:
  @solveloop Method=movavg, Paraml=2 1, Varl=u
