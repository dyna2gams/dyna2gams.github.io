Source: PSOPT Optimal Control Solver, User Manual, Release 4 build 2015 

par:
  V = 2.138

var:
  tf

dyn:
  x y     # state variables
  theta   # control variables
  J       # cost criterion
  
lim:
  0 <= x <= 2
  0 <= y <= 2
  -10 <= theta <= 10
  0 <= tf <= 2
  
t=t0:
  x = 0
  y = 0
  J = 0
  
t=tf:
  x = 1.2
  y = 1.6
  
ini:
  x = linspace(initial(x),final(x))
  y = linspace(initial(y),final(y))

equ:
  x´ == V*cos(theta)
  y´ == V*sin(theta)
  sqr(x-0.4) + sqr(y-0.5) >= 0.1
  sqr(x-0.8) + sqr(y-1.5) >= 0.1
  
obj: 
  minimize tf using nlp
